476  {
477  using (MemoryStream stream = new MemoryStream())
478  {
479  bitmap.Save(stream, System.Drawing.Imaging.ImageFormat.Png); // Was .Bmp, but this did not show a transparent background.
480 
481  stream.Position = 0;
482  BitmapImage result = new BitmapImage();
483  result.BeginInit();
484  // According to MSDN, "The default OnDemand cache option retains access to the stream until the image is needed."
485  // Force the bitmap to load right now so we can dispose the stream.
486  result.CacheOption = BitmapCacheOption.OnLoad;
